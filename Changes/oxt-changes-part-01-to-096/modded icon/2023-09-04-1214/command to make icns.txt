-- this is a replacement icon for OpenXTalk Lite-- I have included the PSD files for editing-- There are MacOS icns files (1024px newer format)-- I've also included an example of how they look in the dock.File: Livecode.icns-- uncompress and place in[Mac:] right-click the app > "Show Package Contents"/Contents/Resources/file: OpenXTalk-lite_1024.png[linux:] /opt/livecode/livecodecommunity-9.6.x/file: OpenXTalk-lite_48.png[linux:] /opt/livecode/livecodecommunity-9.6.x/-- To make icns files on MacOS, you can do it from the terminal if you have a png file ready.-- just name your file "Icon1024.png" on the desktop, and run:cd ~/Desktopmkdir MyIcon.iconsetsips -z 16 16 Icon1024.png --out MyIcon.iconset/icon_16x16.pngsips -z 32 32 Icon1024.png --out MyIcon.iconset/icon_16x16@2x.pngsips -z 32 32 Icon1024.png --out MyIcon.iconset/icon_32x32.pngsips -z 64 64 Icon1024.png --out MyIcon.iconset/icon_32x32@2x.pngsips -z 128 128 Icon1024.png --out MyIcon.iconset/icon_128x128.pngsips -z 256 256 Icon1024.png --out MyIcon.iconset/icon_128x128@2x.pngsips -z 256 256 Icon1024.png --out MyIcon.iconset/icon_256x256.pngsips -z 512 512 Icon1024.png --out MyIcon.iconset/icon_256x256@2x.pngsips -z 512 512 Icon1024.png --out MyIcon.iconset/icon_512x512.pngcp Icon1024.png MyIcon.iconset/icon_512x512@2x.pngiconutil -c icns MyIcon.iconsetrm -R MyIcon.iconset
